server {
    listen              80;
    server_name         ledscreen;
    error_page 502 =200 /e502;

    location / {
        include         uwsgi_params;
        uwsgi_pass      unix:/var/www/webapp/webapp.sock;
    }

    location /static/ {
        gzip_static     on;
        expires         7d;
        add_header      Cache-Control public;
        alias           /var/www/webapp/static/;
    }

    location /favicon.ico {
        access_log      off;
        gzip_static     on;
        add_header      Cache-Control public;
        tcp_nodelay     off;
        expires         max;
        alias           /var/www/webapp/static/favicon.ico;
    }

    location /e502 {
        add_header Content-Type text/plain;
        return 200 "The web application failed to start. Please restart the system. If the issue persists, contact Jacob.";
    }
}
